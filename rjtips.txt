function htmlEncode(str){
	str = str || "";
	return str.replace(/&/g, "&gt;")
			  .replace(/</g, "&lt;")
			  .replace(/>/g, "&gt;")
			  .replace(/\'/g, "&#39;")
			  .replace(/\r\n|\n/g, "<br />")
			  .replace(/^\s+|\s+$/g, "")
			  .replace(/\"/g, "&quot;");
}

function parseTime(s){
	//2010-07-09 15:22:42 +0800
	var arr = s.split(/\s+/);
	var ymd = arr[0].split("-"), hms = arr[1].split(":");
	return new Date(ymd[0], parseInt(ymd[1],10) - 1, ymd[2], hms[0], hms[1], hms[2]).valueOf();
}
